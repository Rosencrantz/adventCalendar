Using load to get some data


var app = $.sammy('#calendar', function() {
  this.use(Sammy.Template);
  
  //Trigger the display of our slide
  this.get('#/:id/slide', function(context) {
    this.trigger('showSlide', {id: this.params.id, path: 'assets/data/slides/advent0.htm'})});
  });

  //load up our slide content and add it into the document
  this.bind('showSlide', function(e, data) {
  	this.load(data['path']).then(function(content) {
  		$('#slide' + data['id'] + ' .slideContent').html(content);
  		$('#slide' + data['id']).addClass('view');
  		$('#modal').show();			
  	});
});
