Basic concepts

  • Sammy applications revolve around the concept of a route.
  • A route consists of 3 arguments: verb, path and callback
  • Routes exist as part of a Sammy application and can be bound to custom events


//Define the application
var app = $.sammy(function() {
    
  //Create a route
  this.get('#/', function() {
    $('body').append('Merry Christmas!');	
  });
});
		  
//run the application with the default route specified
(function() {
  app.run('#/');
})();
