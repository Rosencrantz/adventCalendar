Creating routes


var app = $.sammy(function() {
  this.get('#/', function() {
    //The Default route
  });
    
  this.get('#/:id', function() {
    //Open the advent door identified by the id
  });

  this.route('get', '#/:id/slide', function() {
    //Display the slide for the give date
  });
});

$(function() {
  app.run('#/'); //run the application starting the default route
});
