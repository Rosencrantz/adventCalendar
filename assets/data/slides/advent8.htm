Creating, binding & triggering events


var app = $.sammy(function() {
  this.get('#/:id', function(context) {
    this.trigger('hideSlide', {id: 42}); //Display the advent item
  });
  
  this.bind('hideSlide', function(e, data) {
    //hide an existing slide
  });

  this.bind('webkitTransitionEnd', function(e, data) {
    //Bound to existing event, displays a slide once a door is opened
  });
});
